import sys
import time
import json
import threading
from http.server import BaseHTTPRequestHandler, HTTPServer

try:
    from arduino.app_utils import *
except ImportError:
    print("FATAL ERROR: 'arduino.app_utils' module not found.")
    sys.exit(1)

dashboard_data = {"status": "INITIALIZING", "data": None}

class WebDashboardHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.send_header('Content-type', 'text/html; charset=utf-8')
        self.end_headers()
        html_content = f"""<html>
        <head><meta http-equiv="refresh" content="2"></head>
        <body style="background-color:#121212; color:#00FF00; font-family:monospace; padding:20px;">
            <h1>QITC Secure Edge Adjudicator Dashboard</h1>
            <hr style="border-color:#333;">
            <h2>Latest Result:</h2>
            <pre style="font-size:18px; background:#222; padding:15px;">{json.dumps(dashboard_data, indent=4)}</pre>
        </body>
        </html>"""
        self.wfile.write(html_content.encode('utf-8'))
        
    def log_message(self, format, *args):
        pass

def start_server():
    server = HTTPServer(('0.0.0.0', 8080), WebDashboardHandler)
    server.serve_forever()

def main():
    global dashboard_data
    
    threading.Thread(target=start_server, daemon=True).start()
    print("System: Web Dashboard running. Open http://<Board_IP>:8080 in browser.")
    
    scenarios = [
        {"event_id": "EVT_001", "amount": 5000.0, "currency": "USD", "risk_score": 10},
        {"event_id": "EVT_002", "amount": 15000.0, "currency": "USD", "risk_score": 20},
        {"event_id": "EVT_003", "amount": 8000.0, "currency": "USD", "risk_score": 85}
    ]
    
    try:
        while True:
            for txn_data in scenarios:
                payload = json.dumps(txn_data)
                
                try:
                    response = Bridge.call('adjudicate_txn', payload)
                    if response:
                        dashboard_data = {"status": "SUCCESS", "data": json.loads(response)}
                    else:
                        dashboard_data = {"status": "ERROR", "data": "Empty Response"}
                except Exception as err:
                    dashboard_data = {"status": "BRIDGE_ERROR", "data": str(err)}
                
                time.sleep(3)
                
    except KeyboardInterrupt:
        sys.exit(0)

if __name__ == '__main__':
    main()
