import sys
import time
import json
import socket
import threading
from http.server import BaseHTTPRequestHandler, HTTPServer

def force_print(msg):
    print(msg)
    sys.stdout.flush()

force_print("--- SYSTEM DIAGNOSTIC START ---")

try:
    force_print("Step 1: Checking Network Interface...")
    s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    s.connect(('8.8.8.8', 80))
    current_ip = s.getsockname()[0]
    s.close()
    force_print(f">>> DEVICE IP ADDRESS: {current_ip} <<<")
except Exception as e:
    current_ip = '127.0.0.1'
    force_print(f"Network Error (Using localhost): {e}")

try:
    force_print("Step 2: Loading Arduino Bridge Utilities...")
    from arduino.app_utils import *
    force_print("Bridge Utils Loaded Successfully.")
except ImportError as e:
    force_print(f"FATAL ERROR: 'arduino.app_utils' module not found. ({e})")
    sys.exit(1)

dashboard_data = {"status": "INITIALIZING", "last_update": "N/A", "result": None}

class WebDashboardHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.send_header('Content-type', 'text/html; charset=utf-8')
        self.end_headers()
        html_content = f"""<html>
        <head>
            <meta http-equiv="refresh" content="2">
            <style>
                body {{ background:#0f0f0f; color:#00ff41; font-family:monospace; padding:40px; }}
                .box {{ border:2px solid #00ff41; padding:20px; border-radius:10px; background:#1a1a1a; }}
                h1 {{ color: #fff; }}
                .status {{ font-size: 24px; color: #ffea00; }}
            </style>
        </head>
        <body>
            <h1>Qualcomm QITC - Secure Financial Adjudicator</h1>
            <div class="box">
                <p class="status">System Status: {dashboard_data['status']}</p>
                <hr>
                <h2>Live Transaction Evidence (Meta_A):</h2>
                <pre style="font-size:16px; color:#00ff41;">{json.dumps(dashboard_data['result'], indent=4)}</pre>
            </div>
            <p>Server IP: {current_ip} | Last Update: {time.strftime('%H:%M:%S')}</p>
        </body>
        </html>"""
        self.wfile.write(html_content.encode('utf-8'))
        
    def log_message(self, format, *args):
        pass

def start_server():
    server = HTTPServer(('0.0.0.0', 8080), WebDashboardHandler)
    server.serve_forever()

def main():
    global dashboard_data
    
    force_print(f"Step 3: Starting Web Dashboard at http://{current_ip}:8080")
    threading.Thread(target=start_server, daemon=True).start()
    
    scenarios = [
        {"event_id": "EVT_ALLOW", "amount": 5000.0, "currency": "USD", "risk_score": 10},
        {"event_id": "EVT_DENY_AMT", "amount": 15000.0, "currency": "USD", "risk_score": 20},
        {"event_id": "EVT_PENDING", "amount": 8000.0, "currency": "USD", "risk_score": 85},
        {"event_id": "EVT_DENY_CURR", "amount": 3000.0, "currency": "EUR", "risk_score": 5}
    ]
    
    force_print("Step 4: Beginning Transaction Stream...")
    
    try:
        while True:
            for txn_data in scenarios:
                payload = json.dumps(txn_data)
                force_print(f"[Tx] Transmitting: {payload}")
                
                try:
                    response = Bridge.call('adjudicate_txn', payload)
                    if response:
                        parsed_res = json.loads(response)
                        dashboard_data = {
                            "status": "OPERATIONAL",
                            "result": parsed_res
                        }
                        force_print(f"[Rx] Decision: {parsed_res['status']}")
                    else:
                        dashboard_data["status"] = "MCU_RESPONSE_EMPTY"
                        force_print("[Rx] Error: Empty Response from MCU")
                except Exception as err:
                    dashboard_data["status"] = f"BRIDGE_ERROR"
                    force_print(f"[Bridge Error] {err}")
                
                time.sleep(3)
                
    except KeyboardInterrupt:
        force_print("System shutdown initiated.")
        sys.exit(0)

if __name__ == '__main__':
    main()
