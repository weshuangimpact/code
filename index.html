import sys
import time
import json
import socket
import threading
from http.server import BaseHTTPRequestHandler, HTTPServer

try:
    from arduino.app_utils import *
except ImportError:
    print("FATAL ERROR: 'arduino.app_utils' module not found.")
    sys.exit(1)

# Global data for Dashboard
dashboard_data = {"status": "INITIALIZING", "last_update": "N/A", "result": None}

def get_ip():
    s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    try:
        # Trigger routing table to find the outbound interface
        s.connect(('8.8.8.8', 80))
        ip = s.getsockname()[0]
    except Exception:
        ip = '127.0.0.1'
    finally:
        s.close()
    return ip

class WebDashboardHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(200)
        self.send_header('Content-type', 'text/html; charset=utf-8')
        self.end_headers()
        
        # Simplified UI for HTML Dashboard
        html_content = f"""<html>
        <head>
            <meta http-equiv="refresh" content="2">
            <style>
                body {{ background:#0f0f0f; color:#00ff41; font-family:monospace; padding:40px; }}
                .box {{ border:2px solid #00ff41; padding:20px; border-radius:10px; background:#1a1a1a; }}
                h1 {{ color: #fff; }}
                .status {{ font-size: 24px; color: #ffea00; }}
            </style>
        </head>
        <body>
            <h1>Qualcomm QITC - Secure Financial Adjudicator</h1>
            <div class="box">
                <p class="status">Current System Status: {dashboard_data['status']}</p>
                <hr>
                <h2>Live Transaction Evidence (Meta_A):</h2>
                <pre style="font-size:16px;">{json.dumps(dashboard_data['result'], indent=4)}</pre>
            </div>
            <p>Server IP: {get_ip()} | Last Update: {time.strftime('%H:%M:%S')}</p>
        </body>
        </html>"""
        self.wfile.write(html_content.encode('utf-8'))
        
    def log_message(self, format, *args):
        pass

def start_server():
    server = HTTPServer(('0.0.0.0', 8080), WebDashboardHandler)
    server.serve_forever()

def main():
    global dashboard_data
    
    # 1. Identity Device IP
    current_ip = get_ip()
    print("\n" + "="*50)
    print(f"DEVICE IP ADDRESS: {current_ip}")
    print(f"DASHBOARD URL: http://{current_ip}:8080")
    print("="*50 + "\n")
    
    # 2. Start Background Web Server
    threading.Thread(target=start_server, daemon=True).start()
    
    # 3. Test Scenarios for Rulepack Validation
    scenarios = [
        {"event_id": "EVT_ALLOW", "amount": 5000.0, "currency": "USD", "risk_score": 10},  # Result: O
        {"event_id": "EVT_DENY_AMT", "amount": 15000.0, "currency": "USD", "risk_score": 20}, # Result: X
        {"event_id": "EVT_PENDING", "amount": 8000.0, "currency": "USD", "risk_score": 85},  # Result: -
        {"event_id": "EVT_DENY_CURR", "amount": 3000.0, "currency": "EUR", "risk_score": 5}   # Result: X
    ]
    
    try:
        while True:
            for txn_data in scenarios:
                payload = json.dumps(txn_data)
                print(f"[Tx] Transmitting to MCU: {payload}")
                
                try:
                    # Bridge Call to MCU STM32U585
                    response = Bridge.call('adjudicate_txn', payload)
                    if response:
                        parsed_res = json.loads(response)
                        dashboard_data = {
                            "status": "OPERATIONAL",
                            "result": parsed_res
                        }
                        print(f"[Rx] MCU Decision: {parsed_res['status']}\n")
                    else:
                        dashboard_data["status"] = "MCU_RESPONSE_EMPTY"
                except Exception as err:
                    print(f"[Bridge Error] {err}")
                    dashboard_data["status"] = "BRIDGE_DISCONNECTED"
                
                time.sleep(3) # Wait 3 seconds per cycle
                
    except KeyboardInterrupt:
        print("System shutdown.")
        sys.exit(0)

if __name__ == '__main__':
    main()
