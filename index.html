import sys
import time
import json

try:
    from arduino.app_utils import *
except ImportError:
    print("FATAL ERROR: 'arduino.app_utils' module not found.")
    sys.exit(1)

def main():
    print("System: Initializing Arduino Bridge (MPU-MCU)...")
    
    scenarios = [
        {"event_id": "EVT_001", "amount": 5000.0, "currency": "USD", "risk_score": 10},
        {"event_id": "EVT_002", "amount": 15000.0, "currency": "USD", "risk_score": 20},
        {"event_id": "EVT_003", "amount": 8000.0, "currency": "USD", "risk_score": 85},
        {"event_id": "EVT_004", "amount": 3000.0, "currency": "EUR", "risk_score": 5}
    ]
    
    try:
        while True:
            for txn_data in scenarios:
                payload = json.dumps(txn_data)
                print(f"[Tx] Sending via Bridge: {payload}")
                
                try:
                    response = Bridge.call('adjudicate_txn', payload)
                    
                    if response:
                        print(f"[Rx] MCU Adjudication Result: {response}\n")
                    else:
                        print("[Rx] Error: MCU responded with empty payload.\n")
                        
                except Exception as err:
                    print(f"[Bridge Error] Call failed: {err}\n")
                
                time.sleep(3)
                
    except KeyboardInterrupt:
        print("\nProcess terminated by user.")
        sys.exit(0)

if __name__ == '__main__':
    main()
