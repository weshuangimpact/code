import sys
import time
import json

try:
    import serial
    import serial.tools.list_ports
except ImportError:
    print("ERROR: 'pyserial' module not found. Please install it.")
    sys.exit(1)

SERIAL_PORT = '/dev/ttyS0'
BAUD_RATE = 115200
TIMEOUT_SEC = 2

def scan_ports():
    ports = serial.tools.list_ports.comports()
    print("--- Available Serial Ports ---")
    for p in ports:
        print(f"Device: {p.device} | Description: {p.description}")
    print("------------------------------")
    if not ports:
        print("WARNING: No serial ports detected by OS.")

def init_uart():
    scan_ports()
    try:
        ser = serial.Serial(
            port=SERIAL_PORT,
            baudrate=BAUD_RATE,
            timeout=TIMEOUT_SEC
        )
        print(f"UART Initialized: {SERIAL_PORT} @ {BAUD_RATE}")
        return ser
    except serial.SerialException as e:
        print(f"UART Initialization Error: {e}")
        print("ACTION REQUIRED: Update 'SERIAL_PORT' variable to match one of the available devices listed above.")
        sys.exit(1)

def generate_mock_txn():
    return {
        "event_id": "EVT_001",
        "action": "TXN_TRANSFER",
        "amount": 100000,
        "currency": "USD"
    }

def main():
    uart_conn = init_uart()
    
    try:
        while True:
            txn_data = generate_mock_txn()
            payload = json.dumps(txn_data) + "\n"
            
            uart_conn.write(payload.encode('utf-8'))
            print(f"[Tx] Sent: {payload.strip()}")
            
            response = uart_conn.readline()
            
            if response:
                decoded = response.decode('utf-8').strip()
                print(f"[Rx] Adjudication Result: {decoded}")
            else:
                print("[Timeout] No response from MCU. Check F8/Resume status.")
            
            time.sleep(2)
            
    except KeyboardInterrupt:
        print("\nProcess terminated.")
    finally:
        if uart_conn and uart_conn.is_open:
            uart_conn.close()

if __name__ == '__main__':
    main()
