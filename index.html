QITC MPU - Serial Monitor Version

import sys
import time
import json
from arduino.app_utils import Bridge

def log_section(title):
    print("\n" + "="*70)
    print(f"[ {title} ]")
    print("="*70)
    sys.stdout.flush()

def log_tx(payload):
    print("\n--- TX -------------------------------------------------------")
    print(json.dumps(json.loads(payload), indent=4))
    sys.stdout.flush()

def log_rx(response):
    print("\n--- RX -------------------------------------------------------")
    print(json.dumps(json.loads(response), indent=4))
    sys.stdout.flush()

def main():

    log_section("QITC Secure Financial Adjudicator")
    print("MPU → Bridge → MCU Adjudication Pipeline Active")
    print("System Status: INITIALIZING")
    sys.stdout.flush()

    scenarios = [
        {"event_id": "EVT_ALLOW", "amount": 5000.0, "currency": "USD", "risk_score": 10},
        {"event_id": "EVT_DENY_AMT", "amount": 15000.0, "currency": "USD", "risk_score": 20},
        {"event_id": "EVT_PENDING", "amount": 8000.0, "currency": "USD", "risk_score": 85},
        {"event_id": "EVT_DENY_CURR", "amount": 3000.0, "currency": "EUR", "risk_score": 5}
    ]

    print("\nSystem Status: OPERATIONAL")
    sys.stdout.flush()

    while True:
        for txn in scenarios:

            payload = json.dumps(txn)
            log_tx(payload)

            try:
                response = Bridge.call("adjudicate_txn", payload)

                if response:
                    log_rx(response)
                    print("\nDecision Timestamp:", time.strftime("%H:%M:%S"))
                else:
                    print("\n[ERROR] Empty response from MCU")

            except Exception as e:
                print("\n[BRIDGE ERROR]", str(e))

            print("-"*70)
            sys.stdout.flush()
            time.sleep(3)

if __name__ == "__main__":
    main()
