QITC MPU Dashboard (GUI Version)

import sys
import time
import json
import socket
import threading
import tkinter as tk
from tkinter.scrolledtext import ScrolledText

def force_print(msg):
    print(msg)
    sys.stdout.flush()

force_print("--- SYSTEM DIAGNOSTIC START ---")

try:
    hostname = socket.gethostname()
    current_ip = socket.gethostbyname(hostname)
except:
    current_ip = "127.0.0.1"

try:
    from arduino.app_utils import *
except ImportError as e:
    force_print(f"FATAL ERROR: arduino.app_utils not found ({e})")
    sys.exit(1)

dashboard_data = {
    "status": "INITIALIZING",
    "last_update": "N/A",
    "result": None
}

data_lock = threading.Lock()

root = tk.Tk()
root.title("QITC Secure Financial Adjudicator")
root.geometry("1000x700")
root.configure(bg="#0f0f0f")

title = tk.Label(root,
                 text="Qualcomm QITC - Secure Financial Adjudicator",
                 fg="white",
                 bg="#0f0f0f",
                 font=("Consolas", 24))
title.pack(pady=20)

status_label = tk.Label(root,
                        text="System Status: INITIALIZING",
                        fg="#ffea00",
                        bg="#0f0f0f",
                        font=("Consolas", 20))
status_label.pack(pady=10)

result_box = ScrolledText(root,
                          fg="#00ff41",
                          bg="#1a1a1a",
                          font=("Consolas", 14),
                          height=20)
result_box.pack(fill="both", expand=True, padx=40, pady=20)

def update_ui():
    with data_lock:
        status_label.config(text=f"System Status: {dashboard_data['status']}")
        result_box.delete("1.0", tk.END)
        result_box.insert(tk.END,
                          json.dumps(dashboard_data["result"], indent=4))
    root.after(1000, update_ui)

def worker():
    scenarios = [
        {"event_id": "EVT_ALLOW", "amount": 5000.0, "currency": "USD", "risk_score": 10},
        {"event_id": "EVT_DENY_AMT", "amount": 15000.0, "currency": "USD", "risk_score": 20},
        {"event_id": "EVT_PENDING", "amount": 8000.0, "currency": "USD", "risk_score": 85},
        {"event_id": "EVT_DENY_CURR", "amount": 3000.0, "currency": "EUR", "risk_score": 5}
    ]

    while True:
        for txn in scenarios:
            payload = json.dumps(txn)
            try:
                response = Bridge.call("adjudicate_txn", payload)
                if response:
                    parsed = json.loads(response)
                    with data_lock:
                        dashboard_data["status"] = "OPERATIONAL"
                        dashboard_data["result"] = parsed
                        dashboard_data["last_update"] = time.strftime("%H:%M:%S")
            except:
                with data_lock:
                    dashboard_data["status"] = "BRIDGE_ERROR"
            time.sleep(3)

threading.Thread(target=worker, daemon=True).start()
update_ui()
root.mainloop()
