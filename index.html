import sys
import time
import json
import socket
import threading
from http.server import BaseHTTPRequestHandler, ThreadingHTTPServer

def force_print(msg):
    print(msg)
    sys.stdout.flush()

force_print("--- SYSTEM DIAGNOSTIC START ---")

# ===============================
# Network Detection (Local Only)
# ===============================
try:
    force_print("Step 1: Detecting Local IP...")
    hostname = socket.gethostname()
    current_ip = socket.gethostbyname(hostname)
    force_print(f">>> DEVICE IP ADDRESS: {current_ip} <<<")
except Exception as e:
    current_ip = "127.0.0.1"
    force_print(f"IP Detection Failed. Using localhost. ({e})")

# ===============================
# Bridge Import
# ===============================
try:
    force_print("Step 2: Loading Arduino Bridge Utilities...")
    from arduino.app_utils import *
    force_print("Bridge Utils Loaded Successfully.")
except ImportError as e:
    force_print(f"FATAL ERROR: arduino.app_utils not found ({e})")
    sys.exit(1)

dashboard_data = {
    "status": "INITIALIZING",
    "last_update": "N/A",
    "result": None
}

data_lock = threading.Lock()

# ===============================
# Web Handler
# ===============================
class WebDashboardHandler(BaseHTTPRequestHandler):

    def do_GET(self):
        try:
            with data_lock:
                status = dashboard_data["status"]
                result = dashboard_data["result"]

            self.send_response(200)
            self.send_header("Content-type", "text/html; charset=utf-8")
            self.end_headers()

            html_content = f"""







Qualcomm QITC - Secure Financial Adjudicator




System Status: {status}




Live Transaction Evidence (Meta_A):


{json.dumps(result, indent=4)}





Server IP: {current_ip} | Last Update: {time.strftime('%H:%M:%S')}



"""
            self.wfile.write(html_content.encode("utf-8"))

        except Exception as e:
            force_print(f"[HTTP ERROR] {e}")

    def log_message(self, format, *args):
        return


# ===============================
# Web Server
# ===============================
def start_server():
    try:
        server = ThreadingHTTPServer(("0.0.0.0", 8080), WebDashboardHandler)
        force_print(">>> Web Server STARTED on port 8080")
        server.serve_forever()
    except Exception as e:
        force_print(f">>> Web Server FAILED: {e}")


# ===============================
# Main Loop
# ===============================
def main():
    global dashboard_data

    force_print(f"Step 3: Starting Web Dashboard at http://{current_ip}:8080")

    server_thread = threading.Thread(target=start_server)
    server_thread.start()

    scenarios = [
        {"event_id": "EVT_ALLOW", "amount": 5000.0, "currency": "USD", "risk_score": 10},
        {"event_id": "EVT_DENY_AMT", "amount": 15000.0, "currency": "USD", "risk_score": 20},
        {"event_id": "EVT_PENDING", "amount": 8000.0, "currency": "USD", "risk_score": 85},
        {"event_id": "EVT_DENY_CURR", "amount": 3000.0, "currency": "EUR", "risk_score": 5}
    ]

    force_print("Step 4: Beginning Transaction Stream...")

    try:
        while True:
            for txn_data in scenarios:
                payload = json.dumps(txn_data)
                force_print(f"[Tx] Transmitting: {payload}")

                try:
                    response = Bridge.call("adjudicate_txn", payload)

                    with data_lock:
                        if response:
                            parsed_res = json.loads(response)
                            dashboard_data["status"] = "OPERATIONAL"
                            dashboard_data["result"] = parsed_res
                            dashboard_data["last_update"] = time.strftime("%H:%M:%S")
                            force_print(f"[Rx] Decision: {parsed_res.get('status')}")
                        else:
                            dashboard_data["status"] = "MCU_RESPONSE_EMPTY"
                            force_print("[Rx] Empty response from MCU")

                except Exception as err:
                    with data_lock:
                        dashboard_data["status"] = "BRIDGE_ERROR"
                    force_print(f"[Bridge Error] {err}")

                time.sleep(3)

    except KeyboardInterrupt:
        force_print("System shutdown initiated.")
        sys.exit(0)


if __name__ == "__main__":
    main()
