import sys
import time
import json

try:
    import rpc
except ImportError:
    print("FATAL ERROR: 'rpc' module not found.")
    print("Ensure this script is running inside the official Arduino App Lab container.")
    sys.exit(1)

def generate_mock_txn():
    return {
        "event_id": "EVT_001",
        "action": "TXN_TRANSFER",
        "amount": 100000,
        "currency": "USD"
    }

def main():
    print("System: Initializing MPU-MCU RPC Bridge...")
    
    try:
        while True:
            txn_data = generate_mock_txn()
            payload = json.dumps(txn_data)
            
            print(f"[Tx] Sending to MCU via RPC: {payload}")
            
            try:
                response = rpc.call('adjudicate_txn', payload)
                
                if response:
                    print(f"[Rx] MCU Adjudication Result: {response}")
                else:
                    print("[Rx] MCU responded with empty payload.")
                    
            except Exception as rpc_err:
                print(f"[RPC Error] Call failed or timed out: {rpc_err}")
                print("ACTION REQUIRED: Ensure MCU (STM32U585) has registered the 'adjudicate_txn' RPC function and is running.")
            
            time.sleep(2)
            
    except KeyboardInterrupt:
        print("\nProcess terminated by user.")
        sys.exit(0)

if __name__ == '__main__':
    main()
