import sys
import time
import json

try:
    from arduino.app_utils import *
except ImportError:
    print("FATAL ERROR: 'arduino.app_utils' module not found.")
    print("Ensure you are running this script inside the Arduino App Lab environment.")
    sys.exit(1)

def generate_mock_txn():
    return {
        "event_id": "EVT_001",
        "action": "TXN_TRANSFER",
        "amount": 100000,
        "currency": "USD"
    }

def main():
    print("System: Initializing Arduino Bridge (MPU-MCU)...")
    
    try:
        while True:
            txn_data = generate_mock_txn()
            payload = json.dumps(txn_data)
            
            print(f"[Tx] Sending to MCU via Bridge: {payload}")
            
            try:
                response = Bridge.call('adjudicate_txn', payload)
                
                if response:
                    print(f"[Rx] MCU Adjudication Result: {response}")
                else:
                    print("[Rx] MCU responded with empty payload.")
                    
            except Exception as err:
                print(f"[Bridge Error] Call failed: {err}")
                print("ACTION REQUIRED: Ensure MCU (STM32U585) sketch calls Bridge.provide('adjudicate_txn', ...)")
            
            time.sleep(2)
            
    except KeyboardInterrupt:
        print("\nProcess terminated.")
        sys.exit(0)

if __name__ == '__main__':
    main()
